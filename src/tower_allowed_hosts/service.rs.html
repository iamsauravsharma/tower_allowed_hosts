16/// A layer that validates and allows incoming requests based on their host.
17///
18/// This layer checks the hostname of incoming requests against a list of
19/// allowed hosts. The hostname is resolved using the `Forwarded` header if
20/// `forwarded_token_value` is present and then `Host` header.
21///
22/// If `forwarded_token_value` is specified, the layer will use the `Forwarded`
23/// header to extract the hostname when the specified token-value pair is
24/// present in the header. For example, if `forwarded_token_value` is set to
25/// `("signature", "random_value")`, the layer will parse the `Forwarded` header
26/// and extract the hostname only if the `signature` token matches the value
27/// `random_value`. If the token-value pair is not found, the layer will fall
28/// back to using the `Host` header.
29///
30///  # Example of `Forwarded` Header Parsing
31///
32/// Given the `Forwarded` header:
33/// ```text
34/// for=10.0.10.11;by=10.1.12.11;host=127.0.0.1;signature=random_value,
35/// for=10.0.10.11;by=10.1.12.11;host=127.0.0.3
36/// ```
37/// If `forwarded_token_value` is set to `("signature", "random_value")`, the
38/// layer will extract the host `127.0.0.1` and ignore the other host values.
39///
40/// # Examples
41///
42/// ```rust
43/// let layer = tower_allowed_hosts::AllowedHostLayer::<_, String>::default()
44///     .extend_hosts(vec!["example.com"]);
45/// ```
46#[derive(Clone)]
47pub struct AllowedHostLayer<H, F> {
48    hosts: Vec<H>,
49    forwarded_token_values: Vec<(String, F)>,
50}
51
52impl<H, F> Default for AllowedHostLayer<H, F> {
53    fn default() -> Self {
54        Self {
55            hosts: vec![],
56            forwarded_token_values: vec![],
57        }
58    }
59}

80    /// Extend allowed hosts layer with provided hosts
81    ///
82    /// # Example
83    ///
84    /// ```rust
85    /// let layer = tower_allowed_hosts::AllowedHostLayer::<_, String>::default()
86    ///     .extend_hosts(vec!["example.com"]);
87    /// ```
88    #[must_use]
89    pub fn extend_hosts<I>(mut self, matchers: I) -> Self
90    where
91        I: IntoIterator<Item = H>,
92    {
93        self.hosts.extend(matchers);
94        self
95    }
96
97    /// Add a token-value pair for matching in the `Forwarded` header.
98    ///
99    /// # Examples
100    ///
101    /// ```rust
102    /// let layer = tower_allowed_hosts::AllowedHostLayer::default()
103    ///     .push_host("example.com")
104    ///     .push_forwarded_token_value(("by", "random"));
105    /// ```
106    ///
107    /// In above example only forwarded host name is extracted when host
108    /// forwarded header is present and `by` value matches as `random` as well
109    #[must_use]
110    pub fn push_forwarded_token_value<T, S>(mut self, matcher: T) -> Self
111    where
112        T: Into<(S, F)>,
113        S: Into<String>,
114    {
115        let (token, val) = matcher.into();
116        self.forwarded_token_values
117            .push((Into::<String>::into(token).to_lowercase(), val));
118        self
119    }
120
121    /// Extend allowed hosts layer with provided forwarded header token values
122    ///
123    /// # Example
124    ///
125    /// ```rust
126    /// let layer = tower_allowed_hosts::AllowedHostLayer::default()
127    ///     .push_host("example.com")
128    ///     .extend_forwarded_token_values([["signature", "random"]]);
129    /// ```
130    #[must_use]
131    pub fn extend_forwarded_token_values<I, T, S>(mut self, matchers: I) -> Self
132    where
133        I: IntoIterator<Item = T>,
134        T: Into<(S, F)>,
135        S: Into<String>,
136    {
137        self.forwarded_token_values.extend(
138            matchers
139                .into_iter()
140                .map(Into::into)
141                .map(|(t, v)| (Into::<String>::into(t).to_ascii_lowercase(), v)),
142        );
143        self
144    }
145}

162/// Allowed hosts service that wraps the inner service and validates the request
163/// host.
164#[derive(Clone)]
165pub struct AllowedHost<H, F, S> {
166    inner: S,
167    layer: AllowedHostLayer<H, F>,
168}
169
170impl<H, F, S, ReqBody> Service<Request<ReqBody>> for AllowedHost<H, F, S>
171where
172    S: Service<Request<ReqBody>>,
173    S::Error: Into<BoxError>,
174    H: Matcher,
175    F: Matcher,
176{
177    type Error = BoxError;
178    type Future = AllowedHostFuture<S::Future>;
179    type Response = S::Response;
180
181    fn poll_ready(&mut self, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
182        self.inner.poll_ready(cx).map_err(Into::into)
183    }
184
185    fn call(&mut self, mut req: Request<ReqBody>) -> Self::Future {
186        match get_host(req.headers(), &self.layer.forwarded_token_values) {
187            Ok(host_val) => {
188                let host_allowed = self
189                    .layer
190                    .hosts
191                    .iter()
192                    .any(|host_matcher| host_matcher.matches_value(host_val.as_str()));
193
194                if host_allowed {
195                    req.extensions_mut().insert(Host(host_val.clone()));
196                }
197
198                Self::Future {
199                    response_future: self.inner.call(req),
200                    host: Ok(host_val),
201                    host_allowed,
202                }
203            }
204            Err(err) => {
205                Self::Future {
206                    response_future: self.inner.call(req),
207                    host: Err(err),
208                    host_allowed: false,
209                }
210            }
211        }
212    }
213}

215/// Future for `AllowedHost` service.
216#[pin_project::pin_project]
217pub struct AllowedHostFuture<F> {
218    #[pin]
219    response_future: F,
220    host: Result<String, Error>,
221    host_allowed: bool,
222}
223
224impl<F, Response, E> Future for AllowedHostFuture<F>
225where
226    F: Future<Output = Result<Response, E>>,
227    E: Into<BoxError>,
228{
229    type Output = Result<Response, BoxError>;
230
231    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
232        let this = self.project();
233
234        match (&this.host, &this.host_allowed) {
235            (Ok(allowed_host), true) => {
236                match this.response_future.poll(cx) {
237                    Poll::Ready(result) => {
238                        #[cfg(feature = "tracing")]
239                        tracing::debug!("allowed host: {}", allowed_host);
240                        Poll::Ready(result.map_err(Into::into))
241                    }
242                    Poll::Pending => Poll::Pending,
243                }
244            }
245            (Ok(blocked_host), false) => {
246                #[cfg(feature = "tracing")]
247                tracing::debug!("blocked host: {}", blocked_host);
248                Poll::Ready(Err(
249                    Box::new(Error::HostNotAllowed(blocked_host.clone())).into()
250                ))
251            }
252            (Err(err), _) => Poll::Ready(Err(Box::new(err.clone()).into())),
253        }
254    }
255}

257/// Extract the host from the request headers based on the layer
258/// configuration.
259fn get_host<F>(headers: &HeaderMap, forwarded_token_value: &[(String, F)]) -> Result<String, Error>
260where
261    F: Matcher,
262{
263    let host_header = match extract_from_forwarded(headers, forwarded_token_value)? {
264        Some(host) => host,
265        None => extract_from_host(headers)?,
266    };
267    Ok(host_header)
268}
269
270/// Extract host from `Host` headers.
271fn extract_from_host(headers: &HeaderMap) -> Result<String, Error> {
272    let mut host_headers = headers.get_all(HOST).iter();
273    let first_host = host_headers.next().ok_or(Error::MissingHostHeader)?;
274    if host_headers.next().is_some() {
275        return Err(Error::MultipleHostHeader);
276    }
277    let host_str = String::from_utf8(first_host.as_bytes().to_vec())
278        .map_err(|_| Error::InvalidHostHeader)?
279        .to_string();
280    Ok(host_str)
281}
282
283/// Extract host from `Forwarded` headers only extract host header from allowed
284/// forwarded by values else return None
285fn extract_from_forwarded<F>(
286    headers: &HeaderMap,
287    forwarded_token_values: &[(String, F)],
288) -> Result<Option<String>, Error>
289where
290    F: Matcher,
291{
292    if !forwarded_token_values.is_empty() {
293        for forwarded_header in headers.get_all(FORWARDED) {
294            let header_str = String::from_utf8(forwarded_header.as_bytes().to_vec())
295                .map_err(|_| Error::InvalidForwardedHeader)?;
296            for header in header_str.split(',') {
297                let mut forwarded_host_token = None;
298                let mut token_value_map = HashMap::new();
299                for part in header.split(';') {
300                    if let Some((key, value)) = part.split_once('=') {
301                        let key = key.to_lowercase();
302                        let value = value.to_string();
303
304                        if key == "host" {
305                            forwarded_host_token = Some(value);
306                        } else {
307                            token_value_map.insert(key, value);
308                        }
309                    }
310                }
311                if forwarded_host_token.is_some()
312                    && forwarded_token_values.iter().any(|(token, matcher)| {
313                        token_value_map
314                            .get(token)
315                            .is_some_and(|token_value| matcher.matches_value(token_value))
316                    })
317                {
318                    return Ok(forwarded_host_token);
319                }
320            }
321        }
322    }
323    Ok(None)
